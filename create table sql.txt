 create table users(
    id varchar(4) primary key not null,
    name varchar(20),
    email varchar(50) unique not null,
    phone varchar(12) unique,
    address varchar(100)
);


create table login(
	id varchar(4) primary key not null,
	email varchar(50),
	password varchar(30) not null,
	foreign key(email) references users(email) on delete cascade 
);
